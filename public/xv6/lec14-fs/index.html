// Write b's contents to disk.  Must be locked.
void
bwrite(struct buf *b)
{
  if(!holdingsleep(&b->lock))
    panic("bwrite");
  if (b->blockno >= 32) 
    printf("write: %d\n", b->blockno);
  virtio_disk_rw(b, 1);
}
